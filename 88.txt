-- إنشاء قاعدة بيانات جديدة باسم TestDB
CREATE DATABASE TestDB;
GO

-- الدخول إلى قاعدة البيانات TestDB
USE TestDB;
GO

-- إنشاء جدول تجريبي باسم TestTable
CREATE TABLE dbo.TestTable (
    ID INT,
    FullName VARCHAR(50)
);
GO

-- إدخال بيانات تجريبية داخل الجدول
INSERT INTO dbo.TestTable VALUES
(1, 'Ali'),
(2, 'Ahmed');
GO

-- إنشاء Login على مستوى السيرفر
CREATE LOGIN User2 WITH PASSWORD = 'P@ssw0rd123';
GO

-- إنشاء User داخل قاعدة البيانات وربطه بالـ Login
CREATE USER User2 FOR LOGIN User2;
GO

-- إعطاء صلاحية قراءة SELECT للمستخدم User2 على الجدول
GRANT SELECT ON dbo.TestTable TO User2;
GO

-- عرض بيانات الجدول للتأكد من وجود البيانات
SELECT * FROM dbo.TestTable;
GO

-- سحب صلاحية القراءة SELECT من المستخدم User2
REVOKE SELECT ON dbo.TestTable FROM User2;
GO

-- تغيير كلمة مرور Login
ALTER LOGIN User2 WITH PASSWORD = 'NewSecurePassword123';
GO

-- تعطيل Login ومنعه من تسجيل الدخول
ALTER LOGIN User2 DISABLE;
GO

-- إعادة تفعيل Login
ALTER LOGIN User2 ENABLE;
GO

-- فحص خصائص Login
SELECT name, is_disabled, is_policy_checked, is_expiration_checked
FROM sys.sql_logins
WHERE name = 'User2';
GO

-- معرفة نوع المصادقة
-- 1 = Windows Authentication فقط
-- 0 = Mixed Mode
SELECT SERVERPROPERTY('IsIntegratedSecurityOnly');
GO

-- حذف User من داخل قاعدة البيانات
DROP USER User2;
GO

-- حذف Login من السيرفر
DROP LOGIN User2;
GO
