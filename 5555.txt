3. الحاسبة المتكاملة (حل الواجب: الجمع، الطرح، الضرب)
المهمة: تعديل الحاسبة لتشمل الطرح والضرب كما مطلوب في الواجب المنزلي.  

HTML
<!DOCTYPE html>
<html>
<head>
    <title>Simple Calculator</title>
</head>
<body>
    <h1>Simple Calculator</h1>
    <input type="number" id="num1" placeholder="First number">
    <input type="number" id="num2" placeholder="Second number">
    <br><br>
    <button onclick="addNumbers()">Add</button>
    <button onclick="subtractNumbers()">Subtract</button>
    <button onclick="multiplyNumbers()">Multiply</button>
    <p id="result">Result appears here</p>

    <script>
        function addNumbers() {
            let first = document.getElementById("num1").value;
            let second = document.getElementById("num2").value;
            let sum = Number(first) + Number(second);
            document.getElementById("result").textContent = "Result = " + sum;
        }

        function subtractNumbers() {
            let first = document.getElementById("num1").value;
            let second = document.getElementById("num2").value;
            let sub = Number(first) - Number(second);
            document.getElementById("result").textContent = "Result = " + sub;
        }

        function multiplyNumbers() {
            let first = document.getElementById("num1").value;
            let second = document.getElementById("num2").value;
            let mult = Number(first) * Number(second);
            document.getElementById("result").textContent = "Result = " + mult;
        }
    </script>
</body>
</html>

5. تغيير التنسيق (DOM Styling)

HTML

<html>

    <title>Change Style</title>
</head>

    <script>
        function changeStyle() {
            let heading = document.getElementById("title");
            heading.textContent = "Style Changed";

            heading.style.backgroundColor = "yellow";
        }
    </script>

</html>

7. حل الواجب (تعديل الحاسبة لتشمل الطرح والضرب)
المهمة: بناءً على "Home Work" في الملف الأول، تم دمج العمليات في كود واحد.  

HTML
<!DOCTYPE html>
<html>
<head>
    <title>Full Calculator</title>
</head>
<body>
    <h1>Advanced Calculator</h1>
    <input type="number" id="num1" placeholder="First number">
    <input type="number" id="num2" placeholder="Second number">
    <br><br>
    <button onclick="calculate('add')">Add</button>
    <button onclick="calculate('sub')">Subtract</button>
    <button onclick="calculate('mult')">Multiply</button>
    <p id="result">Result appears here</p>

    <script>
        function calculate(operation) {
            let n1 = Number(document.getElementById("num1").value);
            let n2 = Number(document.getElementById("num2").value);
            let res = 0;
            
            if (operation === 'add') res = n1 + n2;
            else if (operation === 'sub') res = n1 - n2;
            else if (operation === 'mult') res = n1 * n2;
            
            document.getElementById("result").innerHTML = "Result = " + res;
        }
    </script>
</body>
</html>
